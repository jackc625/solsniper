    if ('confirmedTranches' in result) {

      stepSucceeded = chunked.confirmedTranches > 0;

      signature = outcome.signature;
      solReceived = outcome.solReceived;

  } catch (err) {
    lastError = err;  // Track for PUMPPORTAL trigger logic

    log.warn({ mint, step: step.name, message }, 'Sell step failed or timed out -- advancing');

**Add SELL_PARTIAL emission and addSellPrice call for tiered TP partial sells:**

The current SellLadder always transitions to COMPLETED on success. For tiered TP (partial sells), the PositionManager calls `sell()` per tier with a partial token amount. The trade stays in MONITORING between tier fires. However, sell-ladder currently transitions MONITORING -> SELLING -> COMPLETED on every call, which is correct for each tier invocation since PositionManager manages the tier progression.

For incremental P&L tracking, add after the `stepSucceeded` block, before the `transition()` call:

```typescript
if (stepSucceeded) {
  // Check if this is a partial sell (trade was in MONITORING with existing sell_price_sol data).
  // If so, accumulate via addSellPrice and emit SELL_PARTIAL instead of immediately completing.
  const currentTrade = this.tradeStore.getTradeByMint(mint);
  const isPartial = currentTrade?.sellPriceSol != null && currentTrade.sellPriceSol > 0;

  if (isPartial && solReceived != null) {
    // Tiered TP partial: accumulate incrementally, emit SELL_PARTIAL
    this.tradeStore.addSellPrice(mint, solReceived);
    botEventBus.emit('event', {
      type: 'SELL_PARTIAL',
      mint,
      ts: Date.now(),
      detail: `${step.name}: +${solReceived.toFixed(6)} SOL (running total: ${((currentTrade.sellPriceSol ?? 0) + solReceived).toFixed(6)} SOL)`,
      isDryRun: getRuntimeConfig().dryRun,
      pnlSol: solReceived,  // This tier's SOL received
    });
    log.info({ mint, step: step.name, tierSolReceived: solReceived, runningTotal: (currentTrade.sellPriceSol ?? 0) + solReceived }, 'Partial sell confirmed -- SELL_PARTIAL emitted');
  }

  // Complete the trade
  this.tradeStore.transition(mint, 'SELLING', 'COMPLETED', {
    sellSignature: signature,
    sellPriceSol: solReceived,
  });
  // ... rest of SELL_CONFIRMED emission (already in code above)
}
```

Note: The above `isPartial` detection works because for tiered TP, the first tier calls `addSellPrice` which sets `sell_price_sol > 0`. Subsequent tiers detect `sellPriceSol > 0` and emit SELL_PARTIAL. For non-tiered (full) sells, `sellPriceSol` is NULL/0 so `isPartial` is false and only SELL_CONFIRMED fires.

Actually, revise the logic to be clearer: combine the isPartial check with the transition block properly. The full implementation should look like:

```typescript
if (stepSucceeded) {
  // Detect partial sell: if trade already has accumulated sell_price_sol from prior tiers
  const priorTrade = this.tradeStore.getTradeByMint(mint);
  const hasPriorSellPrice = priorTrade?.sellPriceSol != null && priorTrade.sellPriceSol > 0;

  if (hasPriorSellPrice && solReceived != null) {
    // This is a subsequent tier in a tiered TP sequence -- accumulate and emit SELL_PARTIAL
    this.tradeStore.addSellPrice(mint, solReceived);
    const runningTotal = (priorTrade.sellPriceSol ?? 0) + solReceived;
    botEventBus.emit('event', {
      type: 'SELL_PARTIAL',
      mint,
      ts: Date.now(),
      detail: `${step.name}: +${solReceived.toFixed(6)} SOL (total: ${runningTotal.toFixed(6)} SOL)`,
      isDryRun: getRuntimeConfig().dryRun,
      pnlSol: solReceived,
    });
  }

  // Transition SELLING -> COMPLETED with sellPriceSol
  this.tradeStore.transition(mint, 'SELLING', 'COMPLETED', {
    sellSignature: signature,
    sellPriceSol: solReceived,
  });

  const completedTrade = this.tradeStore.getTradeByMint(mint);
  const pnlSol = (completedTrade?.sellPriceSol != null && completedTrade?.amountSol != null)
    ? completedTrade.sellPriceSol - completedTrade.amountSol
    : undefined;
  botEventBus.emit('event', { type: 'SELL_CONFIRMED', mint, ts: Date.now(), detail: step.name, isDryRun: getRuntimeConfig().dryRun, pnlSol });
  log.info({ mint, step: step.name, signature, solReceived, pnlSol }, 'Sell confirmed -- trade COMPLETED');
  return { success: true, step: step.name, signature };
}
```
