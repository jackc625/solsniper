          if (t.pnl_sol > 0) s.wins++;
          else s.losses++;
        }
      }
      return stats;
    }, [history]);

    **Pitfall 5 mitigation:** The current PnlChart recreates the entire chart instance in a useEffect that depends on `[data]`. When the source filter toggles rapidly, this causes chart flickering because destroy+create is expensive.

    **Fix approach:** Keep the chart instance stable across data changes by separating chart creation from data updates.

    1. **Separate chart creation from data update.** Use `useRef` to hold the chart and series instances:
    ```typescript
    const chartRef = useRef<ReturnType<typeof createChart> | null>(null);
    const seriesRef = useRef<ReturnType<ReturnType<typeof createChart>['addSeries']> | null>(null);
    ```

    2. **First useEffect: create chart once** (dependency: `[]`). Create chart, add series, set up ResizeObserver. Store chart and series in refs. Cleanup removes chart:
    ```typescript
    useEffect(() => {
      if (!containerRef.current) return;
      // ... resolve CSS variables ...
      const chart = createChart(containerRef.current, { ... });
      const series = chart.addSeries(LineSeries, { ... });
      chartRef.current = chart;
      seriesRef.current = series;

      const observer = new ResizeObserver(entries => {
        if (entries[0]) {
          const { width } = entries[0].contentRect;
          chart.resize(width, 220);
          chart.timeScale().fitContent();
        }
      });
      observer.observe(containerRef.current);

      return () => {
        observer.disconnect();
        chart.remove();
        chartRef.current = null;
        seriesRef.current = null;
      };
    }, []);
    ```

    3. **Second useEffect: update data** (dependency: `[data]`). Just call `series.setData(data)` and `chart.timeScale().fitContent()`:
    ```typescript
    useEffect(() => {
      if (!seriesRef.current || !chartRef.current) return;
      if (data.length > 0) {
        seriesRef.current.setData(data);
        chartRef.current.timeScale().fitContent();
      } else {
        seriesRef.current.setData([]);
      }

      // Update line color based on final P&L value
      const lastValue = data.length > 0 ? data[data.length - 1]!.value : 0;
      const style = getComputedStyle(document.documentElement);
      const lineColor = lastValue >= 0
        ? (style.getPropertyValue('--green').trim() || '#00ff88')
        : (style.getPropertyValue('--red').trim()   || '#ff4444');
      seriesRef.current.applyOptions({ color: lineColor });
    }, [data]);
    ```

    4. **Keep the same JSX** -- the container div and label remain unchanged.

    This ensures toggling source filter only calls `setData()` on the existing chart instance instead of destroying and recreating it, preventing flickering and memory leaks from un-removed chart instances.
