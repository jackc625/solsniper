      const [results, setResults] = useState<Array<{ tradeId: number; mint: string; status: string }> | null>(null);

      if (!estopDialogOpen.value) return null;

      const canConfirm = input === 'STOP' && !estopExecuting.value;

      const handleConfirm = async () => {
        const result = await triggerEmergencyStop();
        if (result.ok) {
          // Show sell results summary briefly (addresses consensus concern about reporting)
          if (result.sellResults && result.sellResults.length > 0) {
            setResults(result.sellResults);
            // Auto-close after 3s so user sees the results
            setTimeout(() => {
              setResults(null);
              estopDialogOpen.value = false;
              setInput('');
            }, 3000);
          }
          // If no positions to sell, close immediately
          if (!result.sellResults || result.sellResults.length === 0) {
            estopDialogOpen.value = false;
            setInput('');
          }
        } else {
          setError(result.error ?? 'Emergency stop failed -- manually check bot status');
        }
      };

      const handleDismiss = () => {
        if (estopExecuting.value) return; // Don't dismiss while executing
        estopDialogOpen.value = false;
        setInput('');
        setError('');
        setResults(null);
      };

      return (
        <div style={OVERLAY} onClick={handleDismiss}>
          <div style={DIALOG} onClick={(e) => e.stopPropagation()}>
            <div style={DIALOG_TITLE}>CONFIRM EMERGENCY STOP</div>
            <div style={DIALOG_BODY}>
              This will pause all detection and force-sell every open position. Type STOP to confirm.
            </div>
            <input
              type="text"
              placeholder="Type STOP to confirm"
              value={input}
              onInput={(e) => setInput((e.target as HTMLInputElement).value)}
              style={DIALOG_INPUT}
              autoFocus
              disabled={estopExecuting.value}
            />
            {error && <div style={DIALOG_ERROR}>{error}</div>}
            {/* Sell results summary (addresses consensus concern about partial failure reporting) */}
            {results && (
