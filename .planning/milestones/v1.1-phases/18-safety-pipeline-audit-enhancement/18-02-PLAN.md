       function parseIsMutable(data: Buffer): boolean | null {
         let offset = 0;
         if (data.length < 66) return null;
         offset += 1;   // key
         offset += 32;  // updateAuthority
         offset += 32;  // mint
         // name
         if (offset + 4 > data.length) return null;
         const nameLen = data.readUInt32LE(offset); offset += 4 + nameLen;
         // symbol
         if (offset + 4 > data.length) return null;
         const symbolLen = data.readUInt32LE(offset); offset += 4 + symbolLen;
         // uri
         if (offset + 4 > data.length) return null;
         const uriLen = data.readUInt32LE(offset); offset += 4 + uriLen;
         // sellerFeeBasisPoints
         offset += 2;
         // creators Option<Vec<Creator>>
         if (offset >= data.length) return null;
         const hasCreators = data.readUInt8(offset); offset += 1;
         if (hasCreators === 1) {
           if (offset + 4 > data.length) return null;
           const creatorsLen = data.readUInt32LE(offset); offset += 4;
           offset += creatorsLen * 34;  // Each Creator: 32 + 1 + 1 = 34 bytes
         }
         // primarySaleHappened
         if (offset >= data.length) return null;
         offset += 1;
         // isMutable
         if (offset >= data.length) return null;
         return data.readUInt8(offset) === 1;
       }

    function makeMetadataBuffer(opts: { name: string; symbol: string; uri: string; isMutable: boolean; creators?: number }): Buffer {
      // Build sequential Borsh buffer matching Metaplex layout
      const parts: Buffer[] = [];
      parts.push(Buffer.from([4])); // key = MetadataV1
      parts.push(Buffer.alloc(32)); // updateAuthority
      parts.push(Buffer.alloc(32)); // mint
      // name
      const nameBuf = Buffer.from(opts.name);
      const nameLen = Buffer.alloc(4); nameLen.writeUInt32LE(nameBuf.length);
      parts.push(nameLen, nameBuf);
      // symbol
      const symBuf = Buffer.from(opts.symbol);
      const symLen = Buffer.alloc(4); symLen.writeUInt32LE(symBuf.length);
      parts.push(symLen, symBuf);
      // uri
      const uriBuf = Buffer.from(opts.uri);
      const uriLen = Buffer.alloc(4); uriLen.writeUInt32LE(uriBuf.length);
      parts.push(uriLen, uriBuf);
      // sellerFeeBasisPoints
      const fee = Buffer.alloc(2); fee.writeUInt16LE(500);
      parts.push(fee);
      // creators Option
      if (opts.creators !== undefined && opts.creators > 0) {
        parts.push(Buffer.from([1])); // Some
        const cLen = Buffer.alloc(4); cLen.writeUInt32LE(opts.creators);
        parts.push(cLen);
        parts.push(Buffer.alloc(opts.creators * 34)); // creator data

        parts.push(Buffer.from([0])); // None
      }
      // primarySaleHappened
      parts.push(Buffer.from([0]));
      // isMutable
      parts.push(Buffer.from([opts.isMutable ? 1 : 0]));
      return Buffer.concat(parts);
    }
